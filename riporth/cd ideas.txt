#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int change_directory(const char *path) {
    if (path == NULL) {
        fprintf(stderr, "Error: Path cannot be NULL\n");
        return -1;
    }

    // Change the current working directory
    if (chdir(path) != 0) {
        perror("chdir");
        return -1;
    }

    // Optionally, print the new current working directory
    char cwd[1024];
    if (getcwd(cwd, sizeof(cwd)) != NULL) {
        printf("Current working directory changed to: %s\n", cwd);
    } else {
        perror("getcwd");
        return -1;
    }

    return 0;
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "Usage: %s <path>\n", argv[0]);
        return EXIT_FAILURE;
    }

    const char *path = argv[1];
    if (change_directory(path) != 0) {
        return EXIT_FAILURE;
    }

    return EXIT_SUCCESS;
}
